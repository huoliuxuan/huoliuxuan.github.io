#include
#include
#include
using namespace std;
class base //定义一个基类
{protected:
string name; //姓名
string sex; //性别
int age; //年龄
public:
void set(char *name1,char *sex1,int age1)
{
name=name1;
age=age1;
}
void show()
{
cout<<"Base class value:"<<name<<","<<sex<<","<<age<<endl;

}

};
class teacher:public base //基类派生出教师类
{ int sno; //工号
string zc; //职称
double wages; //工资
public:
void set(char *name1,char *sex1,int age1,int sno1,char *zc1,double wages1)
{
base::set(name1,sex1,age1);
sno=sno1;
zc=zc1;
wages=wages1;
}
void show()
{
base::show();
cout<<"Derived class teacher value:"<<sno<<","<<zc<<","<<wages<<endl;
}
};
class student :public base //基类派生出学生类
{ int sno; //学号
string bj; //班级
string zy; //专业
double score; //入学成绩
public:
void set(char *name1,char *sex1,int age1,int sno1,char *bj1,char *zy1,double score1)
{
base::set(name1,sex1,age1);
sno=sno1;
bj=bj1;
zy=zy1;
score=score1;
}
void show()
{
base::show();
cout<<"Derived class student value:"<<sno<<","<<bj<<","<<zy<<","<<score<<endl;
}
};
void main()

{
base b1;
teacher t1;
student s1;
b1.set("王元强","男",42);
b1.show();
t1.set("王元强","男",42,1239,"副教授",6666);
t1.show();
s1.set("王元强","男",42,202014,"20401","医学信息工程",630);
s1.show();
system("pause");
}
